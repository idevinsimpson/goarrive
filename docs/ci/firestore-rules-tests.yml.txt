name: Firestore Rules Tests

# Run on every push and pull request that touches the rules file or the test suite.
# Also runnable manually from the GitHub Actions UI.
on:
  push:
    paths:
      - 'firestore.rules'
      - 'firestore.rules.test.ts'
      - 'functions/package.json'
      - 'functions/tsconfig.json'
  pull_request:
    paths:
      - 'firestore.rules'
      - 'firestore.rules.test.ts'
      - 'functions/package.json'
      - 'functions/tsconfig.json'
  workflow_dispatch:

jobs:
  test-rules:
    name: Run Firestore Security Rules Tests
    runs-on: ubuntu-latest

    steps:
      # ── 1. Checkout ──────────────────────────────────────────────────────────
      - name: Checkout repository
        uses: actions/checkout@v4

      # ── 2. Set up Node.js ────────────────────────────────────────────────────
      - name: Set up Node.js 20
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          cache-dependency-path: functions/package-lock.json

      # ── 3. Install Java (required by Firebase Emulator Suite) ────────────────
      - name: Set up Java 17
        uses: actions/setup-java@v4
        with:
          distribution: 'temurin'
          java-version: '17'

      # ── 4. Install Firebase CLI ───────────────────────────────────────────────
      - name: Install Firebase CLI
        run: npm install -g firebase-tools

      # ── 5. Install functions dependencies (includes @firebase/rules-unit-testing) ──
      - name: Install functions dependencies
        working-directory: functions
        run: npm ci

      # ── 6. Run Firestore rules tests via the emulator ─────────────────────────
      #    firebase emulators:exec starts the Firestore emulator, runs the test
      #    command, then shuts the emulator down automatically.
      - name: Run Firestore rules tests
        working-directory: functions
        run: |
          firebase emulators:exec \
            --only firestore \
            --project demo-goarrive \
            "npm run test:rules"
        env:
          # No real Firebase credentials needed — emulator uses demo project
          FIREBASE_TOKEN: ${{ secrets.FIREBASE_TOKEN }}
